## Overview

The Fluid Dropdown component uses `framer-motion` (or `motion/react`) to create smooth, physics-based animations required for a modern feel. It handles outside clicks using a custom hook.

## Usage

```tsx
import { FluidDropdown } from "@/components/modules/drop-down/Fluid-dropdown";

export function App() {
  return (
    <div className="flex justify-center p-10">
      <FluidDropdown />
    </div>
  );
}
```

## Dependencies

This component relies on the following libraries and hooks:

- `motion/react` or `framer-motion` for animations.
- `lucide-react` for icons.
- `useClickAway` hook for closing the dropdown when clicking outside.

## Hooks

### useClickAway

This hook is essential for the dropdown functionality.

```tsx
import { useEffect, type RefObject } from "react";

export function useClickAway(
  ref: RefObject<HTMLElement>,
  handler: (event: MouseEvent | TouchEvent) => void
) {
  useEffect(() => {
    const listener = (event: MouseEvent | TouchEvent) => {
      if (!ref.current || ref.current.contains(event.target as Node)) {
        return;
      }
      handler(event);
    };

    document.addEventListener("mousedown", listener);
    document.addEventListener("touchstart", listener);

    return () => {
      document.removeEventListener("mousedown", listener);
      document.removeEventListener("touchstart", listener);
    };
  }, [ref, handler]);
}
```
